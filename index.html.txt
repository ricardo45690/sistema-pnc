    function migrateOldData() {
        let changed = false;
        localDB.forEach(r => {
            if (!r.resourceId) {
                const prod = CATALOGO_PRODUCTOS[r.dun];
                if (prod) {
                    r.resourceId = prod.r;
                    changed = true;
                } else {
                    r.resourceId = "MANUAL";
                }
            }
        });
        if (changed) localStorage.setItem('PNC_DB_FINAL', JSON.stringify(localDB));
    }

    function esPastaRestringida(descripcion) {
        if (!descripcion) return false;
        const descUpper = descripcion.toUpperCase();
        return descUpper.includes("MCMACORNI") || descUpper.includes("MACARRON") || descUpper.includes("BARILLA") || descUpper.includes("YERMINA");
    }

    function checkInputManual() {
        const qr = document.getElementById('inputQR').value.trim().toUpperCase();
        const descProducto = document.getElementById('inputDesc').value; 

        if(qr) {
            const isPasta = esPastaRestringida(descProducto);
            const isRI = qr.startsWith("RI"); 

            if (isPasta && !isRI && (qr.startsWith("RC") || qr.startsWith("DT"))) {
                alert("⚠️ ALTO: Este producto (Pastas) DEBE ir en RIDICKA o RIHERDEZ.");
                document.getElementById('inputQR').value = ""; 
                document.getElementById('inputQR').style.borderColor = "var(--neon-pink)";
                document.getElementById('inputIncidencia').innerHTML = "<option value='Sin Especificar'>---</option>";
                return; 
            }

            if (!isPasta && isRI) {
                alert("⚠️ ALTO: La imputabilidad RI es EXCLUSIVA para pastas.");
                document.getElementById('inputQR').value = ""; 
                document.getElementById('inputQR').style.borderColor = "var(--neon-pink)";
                document.getElementById('inputIncidencia').innerHTML = "<option value='Sin Especificar'>---</option>";
                return;
            }

            const sel = document.getElementById('inputIncidencia');
            sel.innerHTML = "";
            const key = Object.keys(damageMap).find(k => qr.includes(k));
            if(key) {
                damageMap[key].forEach(d => sel.innerHTML += `<option>${d}</option>`);
                document.getElementById('inputQR').style.borderColor = "var(--neon-green)";
                document.getElementById('inputQR').style.boxShadow = "0 0 15px var(--neon-green)";
            } else {
                sel.innerHTML = `<option>Input Desconocido</option>`;
                document.getElementById('inputQR').style.borderColor = "var(--neon-pink)";
            }
        }
    }

    function startScan(t) {
        document.getElementById('cameraError').style.display = 'none';
        const readerDiv = document.getElementById('reader');
        readerDiv.style.display = 'block'; 

        if(!html5QrCode) {
            html5QrCode = new Html5Qrcode("reader");
        }

        html5QrCode.start(
            { facingMode: "environment" }, 
            { fps: 10, qrbox: { width: 250, height: 250 } }, 
            (tx) => {
                document.getElementById(t).value = tx; 
                if(t === 'inputDUN') checkProduct(); else checkInputManual();
                
                html5QrCode.stop().then(() => {
                    html5QrCode.clear();
                    html5QrCode = null;
                    document.getElementById('reader').style.display = 'none';
                });
            },
            (errorFrame) => {
                // Ignorar lecturas fallidas por frame
            }
        ).catch(err => {
            console.error("Error nativo de cámara:", err);
            document.getElementById('reader').style.display = 'none';
            document.getElementById('cameraError').style.display = 'block';
        });
    }

    function analyzeSpecificRecurso() {
        const query = document.getElementById('searchRecurso').value.trim().toUpperCase();
        if(!query) return alert("ESCRIBE UN NÚMERO DE RECURSO");

        const db = localDB.filter(r => {
            const resId = r.resourceId || '';
            const dunCode = r.dun || '';
            return resId.includes(query) || dunCode.includes(query);
        });
        
        if(db.length === 0) {
            alert("NO SE ENCONTRARON REGISTROS.");
            document.getElementById('recursoStats').style.display = 'none';
            document.getElementById('recursoEmpty').style.display = 'block';
            return;
        }

        document.getElementById('recursoStats').style.display = 'block';
        document.getElementById('recursoEmpty').style.display = 'none';
        document.getElementById('lblRecurso').innerText = query;

        const total = db.reduce((a,b)=>a+b.cant, 0);
        document.getElementById('recTotal').innerText = total;

        const damageCounts = {};
        db.forEach(r => damageCounts[r.incidencia] = (damageCounts[r.incidencia]||0) + r.cant);
        const sortedDamages = Object.entries(damageCounts).sort((a,b)=>b[1]-a[1]).slice(0, 3); 

        const ctx1 = document.getElementById('chartSpecific').getContext('2d');
        const gradient1 = ctx1.createLinearGradient(0, 0, 400, 0);
        gradient1.addColorStop(0, '#bc13fe'); gradient1.addColorStop(1, '#ff0055'); 
        
        renderChart('chartSpecific', 'bar', sortedDamages.map(x=>x[0]), sortedDamages.map(x=>x[1]), gradient1, true);

        const inputCounts = {};
        db.forEach(r => inputCounts[r.qr] = (inputCounts[r.qr]||0) + r.cant);
        const chartColors = ['#00eaff', '#ffaa00', '#bc13fe', '#00ff99'];
        renderChart('chartVersusRecurso', 'doughnut', Object.keys(inputCounts), Object.values(inputCounts), chartColors, false);
    }

    function cleanOldStorage() {
        for (let i = 0; i < localStorage.length; i++) {
            const key = localStorage.key(i);
            if (key.startsWith('PNC_CATALOGO') && key !== STORAGE_KEY) localStorage.removeItem(key);
        }
    }

    async function updateCatalogFromSheet() {
        document.getElementById('spinnerText').innerText = "SINCRONIZANDO...";
        document.getElementById('loadingSpinner').style.display = 'flex';
        cleanOldStorage();

        try {
            const response = await fetch(CATALOGO_URL);
            if (!response.ok) throw new Error("Error HTTP");
            const data = await response.text();
            const rows = data.split('\n').slice(1);
            const tempCatalog = {};
            
            rows.forEach(row => {
                const c = row.split(',');
                if(c.length >= 2) {
                    const d = c[0].trim(), r = c[1].trim().toUpperCase(), n = c[3]?c[3].replace(/"/g,'').trim():"";
                    if(d.length > 5) tempCatalog[d.padStart(15,'0')] = { n: n, r: r };
                }
            });
            
            localStorage.setItem(STORAGE_KEY, JSON.stringify(tempCatalog));
            CATALOGO_PRODUCTOS = tempCatalog;
            document.getElementById('statusCatalog').innerText = `CATÁLOGO: ${Object.keys(tempCatalog).length} PRODUCTOS`;
            alert(`✅ BD ACTUALIZADA: ${Object.keys(tempCatalog).length} productos.`);
            migrateOldData();
        } catch (error) { alert("❌ ERROR DE RED"); } 
        finally { document.getElementById('loadingSpinner').style.display = 'none'; }
    }

    function checkProduct() {
        const raw = document.getElementById('inputDUN').value.trim();
        const p = CATALOGO_PRODUCTOS[raw.padStart(15,'0')];
        if (p) {
            document.getElementById('dispDesc').innerText = p.n;
            document.getElementById('dispDesc').style.color = "var(--neon-cyan)";
            document.getElementById('dispRecurso').innerHTML = "RECURSO: <span>" + p.r + "</span>";
            document.getElementById('inputDesc').value = p.n;
            document.getElementById('inputRecursoID').value = p.r; 
            
            if(document.getElementById('inputQR').value) checkInputManual();
        } else {
            document.getElementById('dispDesc').innerText = raw.length>8 ? "NO ENCONTRADO" : "ESPERANDO CÓDIGO...";
            document.getElementById('dispDesc').style.color = raw.length>8 ? "var(--neon-pink)" : "#fff";
            document.getElementById('dispRecurso').innerHTML = "RECURSO: <span>---</span>";
            document.getElementById('inputRecursoID').value = "MANUAL";
        }
    }

    function applyFilters() {
        const start = document.getElementById('dateStart').value;
        const end = document.getElementById('dateEnd').value;
        filteredData = localDB.filter(r => (!start || r.dateKey >= start) && (!end || r.dateKey <= end));
        updateUI(filteredData);
    }

    function updateUI(data) {
        document.getElementById('kpiTotal').innerText = data.reduce((a,b)=>a+b.cant,0);
        document.getElementById('kpiRegs').innerText = data.length;

        const counts = {}, damageC = {}, versus = {D:0, H:0};
        
        data.forEach(r => {
            counts[r.qr] = (counts[r.qr]||0) + r.cant;
            damageC[r.incidencia] = (damageC[r.incidencia]||0) + r.cant;
            if(r.qr.includes('DICKA')) versus.D += r.cant;
            if(r.qr.includes('HERDEZ')) versus.H += r.cant;
        });

        renderChart('chartCodes', 'bar', Object.keys(counts), Object.values(counts), ['#00eaff', '#ffaa00', '#bc13fe', '#ff0055']);
        renderChart('chartVersus', 'doughnut', ['DICKA', 'HERDEZ'], [versus.D, versus.H], ['#00eaff', '#ff0055']);
        
        const sortedD = Object.entries(damageC).sort((a,b)=>b[1]-a[1]).slice(0,5);
        renderChart('chartDamage', 'bar', sortedD.map(x=>x[0]), sortedD.map(x=>x[1]), '#00ff99', true);

        document.getElementById('tableBody').innerHTML = data.slice().reverse().map(r => `
            <tr>
                <td><div style="color:#fff; font-size:0.85rem; font-family:monospace;">${r.dun}</div></td>
                <td>
                    <div class="info-recurso" style="font-size:0.75rem;">${r.qr}</div>
                    <div style="color:var(--text-dim); font-size:0.7rem; margin-top:2px;">REC: ${r.resourceId || 'N/A'}</div>
                </td>
                <td><div style="color:var(--text-dim); font-size:0.8rem; font-weight:500;">${r.incidencia}</div></td>
                <td class="text-center"><div style="color:var(--neon-cyan); font-size:1.3rem; font-weight:900;">${r.cant}</div></td>
                <td class="text-center">
                    <button class="btn btn-sm btn-outline-danger" onclick="deleteRow(${r.id})" style="border-radius:50%; width:35px; height:35px; padding:0; line-height:1;">
                        <i class="bi bi-x-lg"></i>
                    </button>
                </td>
            </tr>`).join('');
    }

    function renderChart(id, type, labels, data, colors, horizontal=false) {
        if(charts[id]) charts[id].destroy();
        const scaleOptions = type==='doughnut'?{}:{ 
            x:{ grid:{color:'rgba(255,255,255,0.05)'}, ticks:{color:'#a0a0b0', precision:0, font:{size:10}} }, 
            y:{ grid:{display:false}, ticks:{color:'#fff', precision:0, font:{size:10}} } 
        };
        charts[id] = new Chart(document.getElementById(id), {
            type: type,
            data: { labels: labels, datasets: [{ data: data, backgroundColor: colors, borderRadius: 8, borderWidth:0, hoverOffset: 10 }] },
            options: {
                maintainAspectRatio: false, indexAxis: horizontal?'y':'x',
                plugins: { 
                    legend: {display: type==='doughnut', position:'right', labels:{color:'#fff', usePointStyle:true, font:{size:11}}}, 
                    datalabels: {color:'#fff', font:{weight:'900', size:12}, textShadowColor:'rgba(0,0,0,0.5)', textShadowBlur:5, formatter: (val) => val>0?val:''} 
                },
                scales: scaleOptions,
                layout: { padding: type==='doughnut'?20:0 }
            }
        });
    }

    function saveRecord() {
        const qr = document.getElementById('inputQR').value, dun = document.getElementById('inputDUN').value;
        const cant = document.getElementById('inputCant').value;
        const resourceId = document.getElementById('inputRecursoID').value || "MANUAL"; 
        const desc = document.getElementById('inputDesc').value;
        
        if(!qr || !dun) return alert("❌ FALTAN DATOS.");

        const isPasta = esPastaRestringida(desc);
        const isRI = qr.startsWith("RI");
        if (isPasta && !isRI) return alert("⚠️ ALTO: Las pastas DEBEN ir en RIDICKA o RIHERDEZ.");
        if (!isPasta && isRI) return alert("⚠️ ALTO: Las imputabilidades RI solo son para pastas.");

        document.getElementById('spinnerText').innerText = "GUARDANDO...";
        document.getElementById('loadingSpinner').style.display = 'flex';

        const now = new Date();
        const rec = {
            id: Date.now(), dateKey: getLocalDateString(now), 
            displayDate: `${now.getDate().toString().padStart(2,'0')}/${(now.getMonth()+1).toString().padStart(2,'0')}/${now.getFullYear()}`,
            qr: qr, dun: dun.padStart(15,'0'), desc: desc || "S/N",
            resourceId: resourceId, 
            incidencia: document.getElementById('inputIncidencia').value, cant: parseInt(cant), 
            turno: document.getElementById('inputTurno').value, responsable: document.getElementById('inputUser').value
        };

        localDB.push(rec);
        localStorage.setItem('PNC_DB_FINAL', JSON.stringify(localDB));

        const fd = new FormData();
        fd.append(IDs.fecha, rec.displayDate); fd.append(IDs.imput, rec.qr); 
        fd.append(IDs.dun, `${rec.dun} (REC:${rec.resourceId}) - ${rec.desc}`); 
        fd.append(IDs.dano, rec.incidencia);
        fd.append(IDs.cant, rec.cant); fd.append(IDs.turno, rec.turno); fd.append(IDs.resp, rec.responsable);
        
        fetch(GOOGLE_FORM_URL, {method:'POST', mode:'no-cors', body:fd})
        .then(() => {
            document.getElementById('loadingSpinner').style.display = 'none';
            alert("✅ REGISTRO GUARDADO");
            resetCaptureForm();
            applyFilters();
        })
        .catch(e=>{
            document.getElementById('loadingSpinner').style.display = 'none';
            alert("⚠️ ERROR RED: Guardado Local.");
            resetCaptureForm();
            applyFilters();
        });
    }

    function resetCaptureForm() {
        document.getElementById('inputDUN').value = ""; document.getElementById('inputCant').value = "1";
        document.getElementById('dispRecurso').innerHTML = "RECURSO: <span>---</span>";
        document.getElementById('inputRecursoID').value = "";
        document.getElementById('dispDesc').innerText = "ESPERANDO CÓDIGO...";
        document.getElementById('dispDesc').style.color = "#fff";
        document.getElementById('inputQR').value = "";
        document.getElementById('inputQR').style.borderColor = "rgba(255,255,255,0.1)";
        document.getElementById('inputQR').style.boxShadow = "none";
        document.getElementById('inputIncidencia').innerHTML = "<option value='Sin Especificar'>---</option>";
    }

    async function deleteRow(id) {
        if(confirm("¿ELIMINAR REGISTRO?")) {
            const r = localDB.find(x=>x.id===id);
            if(r) {
                 document.getElementById('spinnerText').innerText = "ELIMINANDO...";
                 document.getElementById('loadingSpinner').style.display = 'flex';
                 const fd = new FormData();
                 fd.append(IDs.dano, "⛔ ELIMINADO"); fd.append(IDs.cant, -r.cant);
                 fd.append(IDs.fecha, r.displayDate); fd.append(IDs.imput, r.qr); fd.append(IDs.dun, r.dun);
                 await fetch(GOOGLE_FORM_URL, {method:'POST', mode:'no-cors', body:fd});
                 document.getElementById('loadingSpinner').style.display = 'none';
            }
            localDB = localDB.filter(x=>x.id!==id);
            localStorage.setItem('PNC_DB_FINAL', JSON.stringify(localDB));
            applyFilters();
        }
    }

    function clearDB() {
        if(prompt("PASSWORD ADMIN:") === ADMIN_PASSWORD && confirm("⚠ SE BORRARÁ EL HISTORIAL LOCAL")) {
            localStorage.clear(); location.reload();
        }
    }

    function switchTab(view, btn) {
        ['view-dashboard','view-capture','view-data','view-recurso'].forEach(v => document.getElementById(v).style.display = 'none');
        document.getElementById('view-'+view).style.display = 'block';
        document.querySelectorAll('.dock-btn').forEach(b => b.classList.remove('active'));
        btn.classList.add('active');
        if(view === 'dashboard' || view === 'data') applyFilters();
    }

    function getLocalDateString(d) { return d.toISOString().split('T')[0]; }
